//显示pdf文件(restful)
	@RequestMapping("/showPdf")
    public void pdfShow(HttpServletResponse response){
		File file = new File(
				"D:/gj/workspace/specialnew/src/main/webapp/js/pdfjs/pdf/demo.pdf");
		byte[] data = null;
		try {
			FileInputStream input = new FileInputStream(file);
			data = new byte[input.available()];
			input.read(data);
			response.getOutputStream().write(data);
			input.close();
		} catch (Exception e) {
			e.printStackTrace();
		}

	}
	
	//跳转到pdf.jsp
	@RequestMapping("toPdf")
	@ResponseBody
	public ModelAndView toPDF(){
		ModelAndView modelAndView = new ModelAndView("page/rcgl/sbsl/pdf");
		return modelAndView;
	}
